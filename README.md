descriptions des classes :

- Clients =
"id_client" : identifiant du client
"nom" : nom du client
"prenom" : prenom du client
"mail" : mail du client
"telephone" : telephone du client
"adresse" : adressse du client

"id_reservation": identiifiant de la reservation
"id_client": identifiant du client qui a reserver
"id_vehicule": identifiant du vehicule
"date_depart": dapte de location
"date_retour": date de fin de location
"forfait_km": forfait kilometrique choisi
"cout_journalier": prix par jour
"prix_km_supp": prix au kilometres suplementaire
"cout_estime": prix estime de la location

- Vehicules =
"id_vehicule": identifiant du vehicule
"marque": marque du vehicule
"modele": modle du vehicule
"cylindree": nombre de cylindre
"kilometrage_actuel": kilometrage a la reservation
"date_mise_en_circulation": date de la premeier mise en circulation
